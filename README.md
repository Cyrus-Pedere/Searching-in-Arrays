4. Searching in Arrays

# Demo Code: Linear search
numbers = [10, 20, 30, 40, 50]
target = 30

found = False
for num in numbers:
    if num == target:
        found = True
        break

if found:
    print(target, "found in the list")
else:
    print(target, "not found in the list")
